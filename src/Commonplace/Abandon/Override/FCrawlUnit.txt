package Commonplace.Abandon.Override;

import Commonplace.Utils.Interfaces.PeculiarityC;
import arc.struct.IntSeq;
import arc.util.io.Reads;
import arc.util.io.Writes;
import mindustry.gen.CrawlUnit;
import mindustry.io.TypeIO;

public class FCrawlUnit extends CrawlUnit implements PeculiarityC {
    protected boolean uploaded = true;
    protected IntSeq pes = new IntSeq();

    public static FCrawlUnit create() {
        return new FCrawlUnit();
    }

    @Override
    public int classId() {
        return 110;
    }

    @Override
    public void update() {
//        if (!uploaded) {
//            UnitPeculiarity.load(this, pes.items);
//            uploaded = true;
//        }
        super.update();
    }

    @Override
    public void read(Reads read) {
        super.read(read);
        pes = TypeIO.readIntSeq(read);
        uploaded = false;
    }

    public void write(Writes write) {
        super.write(write);
        TypeIO.writeIntSeq(write, pes);
    }

    @Override
    public IntSeq getPes() {
        return pes;
    }
}
